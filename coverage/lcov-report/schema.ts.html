import * as v from 'valibot';
 
/**
 * 定义支持的 Valibot Schema 类型及其到 Tablestore 类型的映射（概念上的）。
 * // TODO: v.ObjectSchema<infer TEntries, any> 中的 any 是 Valibot 的限制
 */
export type SupportedValibotSchema =
  | v.StringSchema<any> // STRING
  | v.NumberSchema<any> // INTEGER 或 FLOAT/DOUBLE
  | v.BooleanSchema<any> // BOOLEAN
  | v.BigintSchema<any> // INTEGER (使用 Tablestore.Long 处理)
  | v.BlobSchema<any> // BINARY (Valibot >=0.30 使用 BlobSchema)
  | v.DateSchema<any> // INTEGER (存储为时间戳)
  | v.ArraySchema<any, any> // STRING (存储为 JSON)
  | v.TupleSchema<any, any> // STRING (存储为 JSON)
  | v.ObjectSchema<any, any> // STRING (存储为 JSON)
  | v.RecordSchema<v.AnySchema, v.AnySchema, any>; // STRING (存储为 JSON) - Use v.any() and add third param
// | v.AnySchema; // 允许其他类型，但转换逻辑需要开发者自行处理或扩展
 
/**
 * 从 Valibot ObjectSchema 推断出 Entries (字段定义) 的类型。
 * // TODO: v.ObjectSchema<infer TEntries, any> 中的 any 是 Valibot 的限制
 */
export type InferEntries<T extends v.ObjectSchema<any, any>> = T extends v.ObjectSchema<
  infer TEntries,
  any
>
  ? TEntries
  : never;
 
/**
 * Schema 配置选项。
 */
export interface SchemaOptions {
  /**
   * 主键字段名称列表，必须与 Tablestore 表定义一致。
   */
  primaryKeys: string[];
  /**
   * 是否自动管理时间戳字段 (`createdAt`, `updatedAt`)。
   * 如果为 true，需要在 Schema 定义中包含 `createdAt: v.optional(v.date())` 和 `updatedAt: v.optional(v.date())`。
   */
  timestamps?: boolean;
  /**
   * (可选) 定义 Tablestore 搜索索引 (Search Index)。
   * // TODO: 需要细化索引定义的类型结构，目前为 any。
   */
  indexes?: any;
  // 可以添加其他选项，如版本控制字段名等
}
 
/**
 * Schema 类，封装了 Valibot Schema 定义和 ODM 配置选项。
 */
export class Schema<TSchema extends v.ObjectSchema<any, any>> {
  /** Valibot 对象 Schema 定义 (可能已根据 timestamps 选项调整) */
  public readonly definition: v.ObjectSchema<any, any>; // 调整类型以容纳可能添加的字段
  /** 主键字段名列表 */
  public readonly primaryKeys: string[];
  /** 其他配置选项 */
  public readonly options: SchemaOptions;
 
  constructor(inputDefinition: TSchema, options: SchemaOptions) {
    // 重命名输入参数
    this.options = { ...options }; // 存储选项副本
 
    let finalDefinition = inputDefinition; // 默认使用原始定义
    const initialEntries = { ...inputDefinition.entries }; // 创建原始 entries 副本
 
    // 如果启用了时间戳，调整 Schema 定义
    if (options.timestamps) {
      const entriesWithTimestamps = { ...initialEntries }; // 创建用于添加时间戳的副本
 
      if (!entriesWithTimestamps.hasOwnProperty("createdAt")) {
        console.log("自动为 Schema 添加 'createdAt' 字段 (v.optional(v.date(), () => new Date()))"); // 添加默认值工厂
        entriesWithTimestamps.createdAt = v.optional(v.date(), () => new Date()); // 使用默认值工厂
      } else {
        // 如果用户已定义，打印警告（可能类型不匹配）
        console.warn(
          '警告: Schema 选项启用了 timestamps，但 Schema 定义中已存在 "createdAt" 字段。将使用用户定义的类型。'
        );
      }
 
      if (!entriesWithTimestamps.hasOwnProperty("updatedAt")) {
        console.log("自动为 Schema 添加 'updatedAt' 字段 (v.optional(v.date(), () => new Date()))"); // 添加默认值工厂
        entriesWithTimestamps.updatedAt = v.optional(v.date(), () => new Date()); // 使用默认值工厂
      } else {
        console.warn(
          '警告: Schema 选项启用了 timestamps，但 Schema 定义中已存在 "updatedAt" 字段。将使用用户定义的类型。'
        );
      }
      // 使用包含时间戳的 entries 创建新的 Schema 对象
      // 注意：这里可能需要更复杂的类型处理来保持 TSchema 的精确性，但为了功能先用 any
      finalDefinition = v.object(entriesWithTimestamps) as any; // 使用 any 绕过类型检查，或进行更复杂的类型推断
    }
 
    this.definition = finalDefinition; // 存储最终的（可能已调整的）Schema 定义
 
    // --- 保持原有的校验逻辑，但使用 finalDefinition ---
    // 校验 primaryKeys
    if (!options.primaryKeys || options.primaryKeys.length === 0) {
      throw new Error("Schema 选项中必须定义 primaryKeys。" + options.primaryKeys);
    }
    // 确保所有主键都在最终的 Schema 定义中
    // 注意：finalDefinition 现在可能是新创建的对象
    const finalSchemaKeys = Object.keys(this.definition.entries); // 使用最终定义的 keys
    for (const pk of options.primaryKeys) {
      if (!finalSchemaKeys.includes(pk)) {
        // 抛出错误前检查是否在原始定义中也不存在
        if (!initialEntries.hasOwnProperty(pk)) {
          throw new Error(
            `主键字段 "${pk}" 在 Schema 定义中不存在。可用字段: ${finalSchemaKeys.join(", ")}`
          );
        } else {
          // 这不应该发生，因为 finalDefinition 包含 initialEntries
          console.error(`内部错误：主键 "${pk}" 在初始定义中存在但在最终定义中丢失？`);
          throw new Error(`处理主键 "${pk}" 时发生内部错误。`);
        }
      }
    }
    this.primaryKeys = options.primaryKeys;
    // 原有的时间戳字段存在性警告不再需要，因为我们现在自动添加或保留用户定义的
  }
 
  /**
   * 使用 Valibot Schema 解析和验证输入数据。
   *
   * @param data 输入数据
   * @returns 验证和转换后的数据 (TOutput 类型)
   * @throws {v.ValiError} 如果验证失败
   */
  public parse(data: unknown): v.InferOutput<TSchema> {
    // TODO: 考虑是否需要添加额外的特定于 ODM 的验证逻辑
    return v.parse(this.definition, data);
  }
 
  // 可以添加其他辅助方法，例如：
  // - 获取字段类型
  // - 检查字段是否为主键
  // - ...
 
  /**
   * 获取指定字段的 Valibot Schema 定义。
   * @param fieldName 字段名称
   * @returns 对应的 Valibot Schema 或 undefined
   */
  public getFieldSchema(fieldName: string): unknown | undefined {
    const entries = this.definition.entries as Record<string, unknown>;
    return entries[fieldName];
  }
 
  /**
   * 检查某个字段是否是主键。
   * @param fieldName 要检查的字段名。
   * @returns {boolean} 如果是主键则返回 true，否则返回 false。
   */
  public isPrimaryKey(fieldName: string): boolean {
    return this.primaryKeys.includes(fieldName);
  }
 
  /**
   * 获取主键的定义（字段名和类型）。
   * @returns {Array<{ name: string; type: v.BaseSchema }>} 主键定义数组。
   */
  public getPrimaryKeyDefinitions(): Array<{
    name: string;
    schema: unknown | undefined;
  }> {
    return this.primaryKeys.map((pkName) => ({
      name: pkName,
      schema: this.getFieldSchema(pkName),
    }));
  }
 
  /**
   * 获取所有属性列（非主键）的定义。
   * @returns {Array<{ name: string; type: v.BaseSchema }>} 属性列定义数组。
   */
  public getAttributeDefinitions(): Array<{
    name: string;
    schema: unknown | undefined;
  }> {
    const schemaKeys = Object.keys(this.definition.entries);
    const attributeKeys = schemaKeys.filter((key) => !this.isPrimaryKey(key));
    return attributeKeys.map((attrName) => ({
      name: attrName,
      schema: this.getFieldSchema(attrName),
    }));
  }
 
  /**
   * 安全地解析和验证输入数据。
   * 不会抛出错误，而是返回一个包含结果或错误的对象。
   * @param data 输入数据
   * @returns {v.SafeParseResult<TSchema>} 解析结果对象
   */
  public safeParse(data: unknown): v.SafeParseResult<TSchema> {
    return v.safeParse(this.definition, data);
  }
 
  /**
   * 获取所有属性字段的名称列表（不包括主键）。
   * @returns {string[]} 属性字段名称数组。
   */
  public getAttributeKeys(): string[] {
    const schemaKeys = Object.keys(this.definition.entries);
    return schemaKeys.filter((key) => !this.isPrimaryKey(key));
  }
}
 
/**
 * 创建 Schema 实例的工厂函数。
 *
 * @template TSchema Valibot Object Schema 的类型
 * @param definition Valibot Object Schema 定义
 * @param options Schema 配置选项
 * @returns {Schema<TSchema>} Schema 实例
 */
export function createSchema<TSchema extends v.ObjectSchema<any, any>>(
  definition: TSchema,
  options: SchemaOptions
): Schema<TSchema> {
  // 注意：这里返回的 Schema 实例内部的 definition 类型可能与 TSchema 不同
  // 但外部接口仍然承诺返回 Schema<TSchema>
  return new Schema(definition, options);
}
